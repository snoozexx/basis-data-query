-----------------TABLE------------------
‚Ä¢ Membuat Tabel Baru
--------------------------------------------------------------------------------------------------------------
CREATE TABLE jenis_kendaraan (
  id VARCHAR (10) PRIMARY KEY UNIQUE NOT NULL,
  nama_jenis VARCHAR (50) NOT NULL
);

CREATE TABLE kendaraan_rodhi (
  plat_nomor VARCHAR (8) PRIMARY KEY UNIQUE NOT NULL,
  nama_kendaraan VARCHAR (25) NOT NULL,
  id_jenis_kendaraan VARCHAR (10) NOT NULL,
  keterangan VARCHAR (250),
  kondisi_akhir VARCHAR (5) NOT NULL,
  FOREIGN KEY (id_jenis_kendaraan) REFERENCES jenis_kendaraan(id) ON DELETE CASCADE ON UPDATE CASCADE
);

‚Ä¢ Mengubah Nama Field di Tabel Tertentu
--------------------------------------------------------------------------------------------------------------
ALTER TABLE kendaraan_rodhi CHANGE jeni_kendaraan jenis_kendaraan VARCHAR (10) NOT NULL;

‚Ä¢ Menambahkan Relasi ke Foreign Key
--------------------------------------------------------------------------------------------------------------
ALTER TABLE nama_tabel ADD FOREIGN KEY (id_fk_ditabel) REFERENCES nama_tabel_relasi(id_tabel_relasi);
ALTER TABLE nama_tabel ADD CONSTRAINT nama_lain FOREIGN KEY (id_fk_ditabel) 
    REFERENCES nama_tabel_relasi(id_tabel_relasi);

üìù DML (Data Manipulation Language)


‚Ä¢ Melihat data
--------------------------------------------------------------------------------------------------------------
SELECT * FROM nama_tabel; 
SELECT * FROM nama_tabel WHERE jenis_kendaraan = 'mobil';  
SELECT nama_field, nama_field FROM nama_tabel;  
SELECT mahasiswa.nama_mhs, transaksi_nilai.nilai_akhir 
  FROM transaksi_nilai JOIN mahasiswa ON transaksi_nilai.id_mhs = mahasiswa.Id_mhs;

‚Ä¢ Menambahkan data
--------------------------------------------------------------------------------------------------------------
INSERT INTO kendaraan_rodhi(plat_nomor, nama_kendaraan, jenis_kendaraan, keterangan, kondisi_akhir) VALUES
  ('B7654FWA', 'BMW M4', 'Mobil', NULL, 'Baik'),
  ('N3456FGA', 'BMW M4', 'Mobil', 'Kredit 1 tahun', 'Rusak'),
  ('AA7654KD', 'BMW M3', 'Mobil', NULL, 'Baik');

‚Ä¢ Mengubah/Mengedit Data
--------------------------------------------------------------------------------------------------------------
UPDATE nama_tabel SET nama_kolom = nilai_baru, kolom_lain = nilai_baru WHERE id = 2;

‚Ä¢ Menghapus Data
--------------------------------------------------------------------------------------------------------------
DELETE FROM nama_tabel WHERE plat_nomor = 'B7654FWA';

‚Ä¢ Search Data
--------------------------------------------------------------------------------------------------------------
SELECT * FROM pelanggan WHERE nama = 'Andi'; 
SELECT * FROM pelanggan WHERE nama LIKE '%an%'; 
SELECT * FROM pelanggan WHERE nama LIKE '%an%' AND kota = 'pasuruan'; 
SELECT * FROM pelanggan WHERE nama LIKE '%an%' OR nama LIKE '%ri%';

‚öôÔ∏è Dotnet CLI Commands


--------------- INFO -----------------
dotnet --version                # Cek versi SDK
dotnet --info                   # Info detail SDK & runtime
dotnet --list-sdks              # Daftar SDK
dotnet --list-runtimes          # Daftar runtime

--------------- NEW PROJECT -----------------
dotnet new console -n MyApp     # Console App
dotnet new webapi -n MyApi      # Web API
dotnet new mvc -n MyMvc         # ASP.NET MVC
dotnet new classlib -n MyLib    # Class Library
dotnet new sln -n MySolution    # Solution

--------------- BUILD & RUN -----------------
dotnet build                    # Build project
dotnet run                      # Jalankan aplikasi
dotnet watch run                # Jalankan + reload otomatis
dotnet test                     # Jalankan unit test
dotnet publish -c Release -o out # Publish untuk deploy

--------------- SOLUTION & REFERENCE -----------------
dotnet sln add MyApp/MyApp.csproj
dotnet sln remove MyApp/MyApp.csproj
dotnet add reference ../MyLib/MyLib.csproj
dotnet remove reference ../MyLib/MyLib.csproj

--------------- NUGET PACKAGE -----------------
dotnet add package Nama.Paket
dotnet add package Nama.Paket --version 6.0.0
dotnet remove package Nama.Paket
dotnet restore

--------------- ENTITY FRAMEWORK CORE -----------------
dotnet tool install -g dotnet-ef              # Install EF Tool
dotnet ef migrations add InitialCreate        # Buat migration baru
dotnet ef migrations add UpdateSchema         # Tambah migration baru
dotnet ef migrations remove                   # Hapus migration terakhir
dotnet ef migrations list                     # Daftar migration
dotnet ef database update                     # Apply migration ke DB
dotnet ef database drop                       # Drop database
dotnet ef dbcontext info                      # Info DbContext
dotnet ef dbcontext scaffold "conn" Microsoft.EntityFrameworkCore.SqlServer -o Models
                                              # Generate model dari database

--------------- GLOBAL TOOLS -----------------
dotnet tool list -g
dotnet tool update -g dotnet-ef
dotnet tool uninstall -g dotnet-ef

‚ö° Tips

SQL

  • Gunakan indeks untuk mempercepat query SELECT dengan filter.
  • Selalu tentukan tipe data yang sesuai agar hemat storage.
  • Gunakan FOREIGN KEY untuk menjaga integritas data.

Dotnet CLI

  • Gunakan dotnet watch run saat development biar auto-reload.
  • Simpan dependency dengan dotnet restore jika pindah mesin.
  • Gunakan migration step by step agar skema database selalu sinkron.
